## 01背包
[P1048 [NOIP2005 普及组] 采药](https://www.luogu.com.cn/problem/P1048)
```cpp
#include<bits/stdc++.h>
#define FOR(i,a,b) for(int i=(a);i<=(b);++i)
#define ROF(i,a,b) for(int i=(a);i>=(b);--i)
using namespace std;

int w[105],val[105],dp[1005];

int main(){
    int t,m,res=-1;    
    cin>>t>>m;
    FOR(i,1,m)
        cin>>w[i]>>val[i];
    FOR(i,1,m){
        ROF(j,t,0){
            if(j>=w[i]) dp[j]=max(dp[j-w[i]]+val[i],dp[j]);
        }
    }    
    cout<<dp[t];
    return 0;
}
```
